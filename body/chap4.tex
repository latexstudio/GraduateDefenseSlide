\begin{frame}[c]{POI空间分析(同位模式)}
    \alert{关联规则算法}

    \vspace{1em}
    \begin{itemize}
        \item 事务项$T$，由集合$I_1,I_2,\ldots,I_m$组成
        \item $P \in T, Q \in T$，规则$P \rightarrow Q$
        \item 置信度约束 $c(P \rightarrow Q) \ge min\_c$
        \item 支持度约束 $s(P \rightarrow Q) \ge min\_s$
    \end{itemize}

    \pause
    \vspace{1em}
    \alert{Apriori算法}

    \begin{itemize}
        \item 频繁项的子 集必须也是频繁的
        \item 非频繁项的超集必定非频繁
    \end{itemize}
\end{frame}

        \begin{column}{0.7 \textwidth}
            \alert{参与率}
            \begin{equation}
                PR(c,f_i)=\frac{|\pi_{f_i}(table\_instance(c))|}{|table\_instance(\{f_i\})|}
            \end{equation}
            $PR(\{A,B,C\},A)=2/4=0.5$
            
            $PR(\{A,B,C\},B)=2/5=0.4$
            
            $PR(\{A,B,C\},C)=2/3=0.67$

            \pause
            \vspace{0.5em}
            \alert{参与度}
            \begin{equation}
                PI(c)=min_{i=1}^{k}\{PR(c,f_i)\}
            \end{equation}

            $PI(\{A,B,C\})=min(0.5,0.4,0.67)=0.4$
        \end{column}
    \end{columns}
\end{frame}
